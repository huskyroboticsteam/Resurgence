201 DataPoint(datatime_t time, T data) : datapoint({time, data}) {}
202
208 operator bool() const {
209 return isValid();
210 }
211
219 bool isFresh(std::chrono::milliseconds duration) const {
220 return isValid() && dataclock::now() - duration <= getTime();
221 }
222
228 bool isValid() const {
229 return datapoint.has_value();
230 }
231
238 datatime_t getTime() const {
239 return datapoint.value().first;
240 }
241
248 T getData() const {
249 if (isValid()) {

251 } else {
252 throw bad_datapoint_access();
253 }
254 }
255
263 T getDataOrElse(T defaultData) {
264 return isValid() ? getData() : defaultData;
265 }

276 template <typename F>
277 DataPoint<std::invoke_result_t<F, T>> transform(const F& f) {
278 if (isValid()) {

280 } else {
281 return {};
282 }
283 }
